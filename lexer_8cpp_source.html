15  void lexer::getCh() {
16  start:
17  if (!stream) {
18  throw endOfFileException();
19  }
20  if (!stream.get(curCh)) {
21  curCh = '\0';
22  }
23  if (curCh == L'\n') {
24  line++, col = 0;
25  } else if (curCh == L'\r') {
26  goto start;
27  } else {
28  col++;
29  }

32  lexer::token lexer::scan() {
33  if (curCh == '\0') {
34  return {line, col, token::tokenKind::eof};
35  }
36  while (curCh == ' ' or curCh == '\n' or curCh == '\r' or curCh == '\t') getCh();
37  if (std::isalpha(curCh) or curCh == '_') {
38  return curToken = alphaStart();
39  } else if (std::isdigit(curCh)) {
40  return curToken = digitStart();
41  } else if (curCh == L'+') {
42  return curToken = plusStart();
43  } else if (curCh == L'-') {
44  return curToken = minusStart();
45  } else if (curCh == L'*') {
46  return curToken = asteriskStart();
47  } else if (curCh == L'/') {
48  return curToken = slashStart();
49  } else if (curCh == L'%') {
50  return curToken = percentSignStart();
51  } else if (curCh == L'!') {
52  return curToken = notStart();
53  } else if (curCh == L'=') {
54  return curToken = equalStart();
55  } else if (curCh == L'>') {
56  return curToken = greaterStart();
57  } else if (curCh == L'<') {
58  return curToken = lessStart();
59  } else if (curCh == L'"' or curCh == L'\'') {
60  return curToken = strStart();
61  } else if (curCh == L';') {
62  return curToken = semicolonStart();
63  } else if (curCh == L':') {
64  return curToken = colonStart();
65  } else if (curCh == L',') {
66  return curToken = commaStart();
67  } else if (curCh == L'.') {
68  return curToken = dotStart();
69  } else if (curCh == L'(') {
70  return curToken = leftParenthesesStart();
71  } else if (curCh == L')') {
72  return curToken = rightParenthesesStart();
73  } else if (curCh == L'[') {
74  return curToken = leftBracketStart();
75  } else if (curCh == L']') {
76  return curToken = rightBracketStart();
77  } else if (curCh == L'{') {
78  return curToken = leftBracesStart();
79  } else if (curCh == L'}') {
80  return curToken = rightBracesStart();
81  } else if (curCh == L'&') {
82  return curToken = andStart();
83  } else if (curCh == L'|') {
84  return curToken = orStart();
85  } else if (curCh == L'^') {
86  return curToken = xorStart();
87  } else if (curCh == L'\0') {
88  return curToken = {line, col, token::tokenKind::eof, false};
89  } else {
90  throw unexpectedTokenException(line, col, curCh);
91  }

151  lexer::token lexer::strStart() {
152  vchar strV = curCh;
153  lexer::token tok{line, col, token::tokenKind::string, vstr()};
154  getCh();
155  while (curCh != strV) {
156  if (curCh == '\\') {
157  getCh();
158  tok.strVal += '\\';
159  }
160  tok.strVal += curCh;
161  getCh();
162  }
163  getCh(); // skip "
164  std::wistringstream ss{tok.strVal};
165  tok.strVal = {};
166  parseString(ss, tok.strVal);
167  return tok;
