79  interpreter::invokeFunc(managedPtr<value> func, const vec<value> &args, const managedPtr<value> &passThisPtr) {
80  if (func->isRef())
81  func = func->refObj;
82  try {
83  switch (func->kind) {
84  case value::vKind::vFunc: {
85  stack.push_back({env->dumpRuntimeSourceFileMsg(func->getFunc()), func->getFunc().moduleCxt, {}});
86  stack.back().pushLocalCxt({});
87 
88  if (passThisPtr)
89  stack.back().localCxt.back()[L"this"] = passThisPtr;
90 
91  for (vsize i = 0; i < args.size(); i++) {
92  if (i < args.size()) {
93  stack.back().localCxt.back()[func->getFunc().argsName[i]] = managePtr(args[i]);
94  } else {
95  stack.back().localCxt.back()[func->getFunc().argsName[i]] = managePtr(value{});
96  }
97  }
98  interpret(func->funcObj->code);
99  stack.pop_back();
100  return {};
101  }
102  case value::vKind::vLambda: {
103  stack.push_back(
104  {env->dumpRuntimeSourceFileMsg(func->getLambda()), func->getLambda().func.moduleCxt, {}});
105  stack.back().pushLocalCxt({});
106 
107  stack.back().localCxt.back()[L"outer"] = func->getLambda().outerCxt;
108 
109  if (passThisPtr)
110  stack.back().localCxt.back()[L"this"] = passThisPtr;
111 
112  for (vsize i = 0; i < func->getLambda().func.argsName.size(); i++) {
113  if (i < args.size()) {
114  stack.back().localCxt.back()[func->getLambda().func.argsName[i]] = managePtr(args[i]);
115  } else {
116  stack.back().localCxt.back()[func->getLambda().func.argsName[i]] = managePtr(value{});
117  }
118  }
119  interpret(func->getLambda().func.code);
120  stack.pop_back();
121  return {};
122  }
123  case value::vKind::vNativeFuncPtr: {
124  return (*func->nativeFuncObj)((void *) {this}, args, passThisPtr);
125  }
126  default: {
127  if (auto it = func->members.find(L"rexInvoke"); it != func->members.end()) {
128  return invokeFunc(it->second, args, func);
129  } else {
130  throw signalException(makeErr(L"internalError", L"not a invoke-able object"));
131  }
132  }
133  }
134  } catch (signalReturn &e) {
135  stack.pop_back();

147  value interpreter::interpret(const AST &target) {
148  switch (target.kind) {
149  case AST::treeKind::basicLiterals: {
150  switch (target.leaf.kind) {
151  case lexer::token::tokenKind::integer:
152  return {target.leaf.basicVal.vInt};
153  case lexer::token::tokenKind::decimal:
154  return {target.leaf.basicVal.vDeci};
155  case lexer::token::tokenKind::boolean:
156  return {target.leaf.basicVal.vBool};
157  case lexer::token::tokenKind::string:
158  return {target.leaf.strVal, stringMethods::getMethodsCxt()};
159  case lexer::token::tokenKind::kNull:
160  default:
161  return {};
162  }
163  }
164  case AST::treeKind::listLiteral: {
165  value::vecObject obj{};
166  for (auto &i: target.child) {
167  obj.push_back(managePtr(interpret(i)));
168  }
169  return {obj, rex::vecMethods::getMethodsCxt()};
170  }
171  case AST::treeKind::objectLiteral: {
172  value::cxtObject cxt;
173  value r;
174  for (auto &i: target.child) {
175  r = interpret(i.child[1]);
176  cxt[i.child[0].leaf.strVal] = managePtr(r.isRef() ? r.getRef() : r);
177  }
178  return {cxt};
179  }
180  case AST::treeKind::identifier: {
181  auto &curModCxt = stack.empty() ? moduleCxt : stack.back().moduleCxt;
182  // 优先从栈中查找
183  if (!stack.empty()) {
184  for (auto it = stack.back().localCxt.rbegin();
185  it != stack.back().localCxt.rend(); it++) {
186  if (auto vit = it->find(target.leaf.strVal); vit != it->end()) {
187  return {vit->second->isRef() ? vit->second->refObj : vit->second};
188  }
189  }
190  }
191  // 全局范围解释
192  if (auto vit = interpreterCxt.find(target.leaf.strVal);
193  vit != interpreterCxt.end()) {
194  // 在当前Module Context中查找
195  return {vit->second->isRef() ? vit->second->refObj : vit->second};
196  }
197  if (auto vit = curModCxt->members.find(target.leaf.strVal);
198  vit != curModCxt->members.end()) {
199  // 在当前Module Context中查找
200  return {vit->second->isRef() ? vit->second->refObj : vit->second};
201  }
202  if (auto vit = env->globalCxt->members.find(target.leaf.strVal);
203  vit != env->globalCxt->members.end()) {
204  // 在Global Context中查找
205  return {vit->second->isRef() ? vit->second->refObj : vit->second};
206  }
207  // 符号不存在
208  throw signalException(makeErr(L"internalError", (vstr)(env->dumpRuntimeSourceFileMsg(stack.empty() ? moduleCxt : stack.back().moduleCxt, target)) + L": undefined symbol: `" + target.leaf.strVal + L"`"));
209  }
210  case AST::treeKind::subscriptExpression: {
211  // 处理最左下的subscript，在memberExpression右端的交给memberExpression处理分支
212  value val = interpret(target.child[0]);

214  value r = interpret(target.child[1]);
215  if (r.isRef())
216  getSelfRef(r);
217  switch (l.kind) {
218  case value::vKind::vStr:
219  return {(vint) {(*l.strObj)[r.getInt()]}};
220  case value::vKind::vVec: {
221  return {(*l.vecObj)[r.getInt()]->isRef() ? (*l.vecObj)[r.getInt()]->refObj
222  : (*l.vecObj)[r.getInt()]};
223  }
224  default: {
225  if (auto it = l.members.find(L"rexIndex"); it != l.members.end())
226  return invokeFunc(it->second, {r}, val.isRef() ? val.refObj : managePtr(l));
227  if (r.kind == value::vKind::vStr)
228  return l[r.getStr()];

230  throw signalException(makeErr(L"internalError", (vstr)(env->dumpRuntimeSourceFileMsg(stack.empty() ? moduleCxt : stack.back().moduleCxt, target)) + L": not a subscript-able object"));
231  }
232  }
233  }

243  }
244  case AST::treeKind::lambdaDefinition: {

268  }

283  value{func});
