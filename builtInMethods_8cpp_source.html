519  vchar *formatterTagInfo::parse(vchar *str) {
520  vstr a;
521  vstr b;
522 
523  while (isalpha(*str)) if (*str == L'\0') { return nullptr; } else a += *(str++);
524  if (a == L"str") {

526  } else if (a == L"int") {
527  kind = tagKind::vInt;
528  } else if (a == L"deci") {
529  kind = tagKind::vDeci;
530  } else if (a == L"bool") {
531  kind = tagKind::vBool;
532  } else if (a == L"dol") {
533  kind = tagKind::dollar;
534  } else {
535  return nullptr;
536  }
