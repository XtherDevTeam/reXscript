8  void ffiDescriptor::fromFile(const vstr &path) {
9  // read file
10  vbytes u8Path = wstring2string(path);
11  FILE *fp = fopen(u8Path.c_str(), "r+");
12  if (!fp)
13  throw std::runtime_error("ffiDescriptor::fromFile: fp == nullptr");
14  fseek(fp, 0, SEEK_END);
15  long pos = ftell(fp);
16  fseek(fp, 0, SEEK_SET);
17  vbytes buf(pos, vchar{});
18  fread(buf.data(), pos, 1, fp);
19  fclose(fp);
20  std::wstringstream ffiFile(string2wstring(buf));
21  buf.clear();
22 
23  // parse file
24  // 0 is global, 2 is types, 3 is types-parsing, 5 is exports, 7 is helpers, 8 is consts
25  int mode{};
26  vstr parsingTemp{};
27 
28  vstr line{};
29  for (std::getline(ffiFile, line); ffiFile; std::getline(ffiFile, line)) {
30  if (line.empty())
31  continue;
32  line.erase(0, line.find_first_not_of(L' '));
33  line.erase(line.find_last_not_of(L' ') + 1);
34 
35  switch (mode) {
36  case 0: {
37  // global
38  if (line.starts_with(L"ffi-lib: ")) {
39  lib = line.substr(9);
40  } else if (line.starts_with(L"ffi-types:")) {
41  mode = 2;
42  } else if (line.starts_with(L"ffi-exports:")) {
43  mode = 5;
44  } else if (line.starts_with(L"ffi-consts:")) {
45  mode = 8;
46  } else if (line.starts_with(L"ffi-helpers")) {
47  mode = 7;
48  } else {
49  throw std::runtime_error("ffiDescriptor::fromFile: invalid ffi config");
50  }
51  break;
52  }
53  case 2: {
54  if (line.starts_with(L":end-ffi-types")) {
55  mode = 0;
56  break;
57  }
58  // types
59  parsingTemp = line.substr(0, line.length() - 1);
60  mode = 3;
61  break;
62  }
63  case 3: {
64  // type-parsing
65  if (line.starts_with(L"c-style: ")) {
66  ffiTypes[parsingTemp].cStyle = line.substr(9);
67  } else if (line.starts_with(L"type-to-rex: ")) {
68  ffiTypes[parsingTemp].typeToRex = line.substr(13);
69  } else if (line.starts_with(L"rex-to-type: ")) {
70  ffiTypes[parsingTemp].rexToType = line.substr(13);
71  } else if (line.starts_with(L':' + parsingTemp)) {
72  mode = 2;
73  }
74  break;
75  }
76  case 5: {
77  // exports-parsing
78  if (line.starts_with(L"name: ")) {
79  parsingTemp = line.substr(6);
80  } else if (line.starts_with(L"symbol: ")) {
81  ffiExports[parsingTemp].name = line.substr(8);
82  } else if (line.starts_with(L"params: ")) {
83  split(line.substr(8), vstr{L","}, ffiExports[parsingTemp].params);
84  for (auto &i: ffiExports[parsingTemp].params) {
85  i.erase(0, i.find_first_not_of(L' '));
86  i.erase(i.find_last_not_of(L' ') + 1);
87  }
88  } else if (line.starts_with(L"result: ")) {
89  ffiExports[parsingTemp].resultType = line.substr(8);
90  } else if (line.starts_with(L":end-ffi-exports")) {
91  mode = 0;
92  }
93  break;
94  }
95  case 7: {
96  if (line.starts_with(L":end-ffi-helpers")) {
97  mode = 0;
98  } else {
99  helpers += line + L"\n";
100  }
101  break;
102  }
103  case 8: {
104  // exports-parsing
105  if (line.starts_with(L"name: ")) {
106  parsingTemp = line.substr(6);
107  } else if (line.starts_with(L"symbol: ")) {
108  ffiConsts[parsingTemp].name = line.substr(8);
109  } else if (line.starts_with(L"type: ")) {
110  ffiConsts[parsingTemp].type = line.substr(6);
111  } else if (line.starts_with(L"value: ")) {
112  ffiConsts[parsingTemp].value = line.substr(7);
113  } else if (line.starts_with(L":end-ffi-consts")) {
114  mode = 0;
115  }
116  break;
117  }
118  default: {
119  break;
120  }
121  }
122  }
