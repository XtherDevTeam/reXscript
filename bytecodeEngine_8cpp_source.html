21  bytecodeModule compile(const vstr &filepath) {
22  bytecodeModule result;
23  uint64_t entryBlock = result.putCodeStruct({});
24  std::ifstream f(wstring2string(filepath), std::ios::in);
25  if (f.is_open()) {
26  f.seekg(0, std::ios::end);
27  long fileLen = f.tellg();
28  vbytes buf(fileLen, vbyte{});
29  f.seekg(0, std::ios::beg);
30  f.read(buf.data(), fileLen);
31  std::wstringstream ss(string2wstring(buf));
32  buf.clear();
33 
34  rex::lexer lexer{ss};
35  rex::parser parser{lexer};
36  rex::AST ast = parser.parseFile();
37  bytecodeEngine::codeBuilder builder(result, *result.codeStructs[entryBlock]);
38  for (auto &i: ast.child) {
39  builder.buildStmt(i);
40  }
41  result.entryBlock = entryBlock;
42  for (auto &i: result.codeStructs) {
43  i->msg.file = filepath;
44  }
45  return result;
46  } else {
47  throw importError(L"Cannot open file: file not exist or damaged");
48  }
49  }

59  void rexThreadWrapper(managedPtr<environment> env, vint tid, managedPtr<value> cxt, managedPtr<value> func,
60  vec<value> args, managedPtr<value> passThisPtr) {
61  auto it = managePtr(interpreter{env, {}, cxt});
62 
63  it->interpreterCxt[L"thread_id"] = managePtr(value{tid});
64  try {
65  auto res = it->invokeFunc(func, args, passThisPtr);
66  env->threadPool[tid].setResult(managePtr(res.isRef() ? res.getRef() : res));
67  } catch (rex::signalException &e) {
68  std::cerr << "Uncaught exception in thread " << tid << ": " << rex::wstring2string((rex::value) e.get())
69  << std::endl;

71  throw;
72  } catch (rex::rexException &e) {
73  std::cerr << "Uncaught exception in thread " << tid << ": " << e.what() << std::endl;
74  std::cerr << wstring2string(it->getBacktrace()) << std::endl;
75  throw;
76  } catch (std::exception &e) {
77  std::cerr << "Uncaught exception in thread " << tid << ": " << e.what() << std::endl;
78  std::cerr << wstring2string(it->getBacktrace()) << std::endl;
79  throw;
80  }
